# VCC Sesión 3 — M2 Proyectos + M3 Gobernanza — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Agregar al VCC el área principal con dos vistas (M2 Proyectos + M3 Gobernanza), ejecución de scripts en tiempo real vía WebSocket, acción de abrir VS Code por ambiente, y estado real de túneles SSH en el sidebar.

**Architecture:** El backend extiende el HTTP server de Express para adjuntar un WebSocketServer en el mismo puerto 8080. Tres rutas nuevas cubren túneles (TCP check), proyectos (spawn VS Code) y gobernanza (spawn pwsh, stream via WS). El frontend agrega navegación por tabs, renderizado de proyectos desde el registry, panel de output con WebSocket client, y polling de estado real de túneles.

**Tech Stack:** Node.js 18+ (ES modules), Express 4, ws 8.x, Vanilla JS, CSS Custom Properties, JetBrains Mono

## Global Constraints

- ES modules en todo el backend (`"type": "module"` en package.json) — usar `import/export`, nunca `require()`
- `WORKSPACE_ROOT = 'C:\\Users\\Carlos Valera\\OneDrive\\Escritorio\\AI-Workspace'` — no hardcodear en rutas nuevas, importar de config.js

- `spawn('code', [...], { shell: true })` en Windows — `shell: true` obligatorio
- `spawn('pwsh', ['-NoProfile', '-File', ...])` para scripts de gobernanza
- Frontend: vanilla JS puro — sin frameworks, sin npm en frontend
- CSS: todas las variables de color desde `:root` — no hardcodear hex en reglas nuevas
- Git: un commit por tarea al finalizar

```js
import { Router } from 'express';
import net from 'net';
import { TUNNEL_PORTS } from '../config.js';

const router = Router();

function checkPort(port) {
  return new Promise((resolve) => {
    const socket = net.createConnection({ port, host: '127.0.0.1' });
    socket.setTimeout(500);
    socket.on('connect',  () => { socket.destroy(); resolve(true);  });
    socket.on('error',    () =>                     resolve(false)  );
    socket.on('timeout',  () => { socket.destroy(); resolve(false); });
  });
}

router.get('/', async (_req, res, next) => {
  try {
    const results = await Promise.all(
      TUNNEL_PORTS.map(async (port) => [String(port), await checkPort(port)])
    );
    res.json(Object.fromEntries(results));
  } catch (err) {
    next(err);
  }
});

export default router;
```

- [ ] **Step 1: Implementar projects.js**

Reemplazar `backend/routes/projects.js`:

```js
import { Router } from 'express';
import { readFile } from 'fs/promises';
import { spawn } from 'child_process';
import { PATHS } from '../config.js';

const router = Router();

router.post('/:id/environments/:env/open-vscode', async (req, res, next) => {
  const { id, env } = req.params;

  let registry;
  try {
    registry = JSON.parse(await readFile(PATHS.registry, 'utf8'));
  } catch {
    return res.status(500).json({ error: 'No se pudo leer el registry' });
  }

  const project = registry.projects.find(p => p.id === id);
  if (!project) {
    return res.status(404).json({ error: `Proyecto '${id}' no encontrado` });
  }

  const environment = (project.environments || []).find(e => e.name === env);
  if (!environment) {
    return res.status(404).json({ error: `Ambiente '${env}' no encontrado en '${id}'` });
  }

  const { host, remotePath } = environment;
  if (!host || !remotePath) {
    return res.status(400).json({ error: 'host o remotePath no definido para este ambiente' });
  }

  try {
    const child = spawn('code', ['--remote', `ssh-remote+${host}`, remotePath], {
      shell: true,
      detached: true,
      stdio: 'ignore',
    });
    child.unref();
    res.json({ ok: true });
  } catch (err) {
    next(err);
  }
});

export default router;
```

### Task 4: Backend — POST /api/govern/run + WebSocket streaming

**Files:**
- Modify: `backend/routes/govern.js` (reemplazar stub de Task 1)

**Interfaces:**
- Consumes: `wss` (WebSocketServer de server.js, recibido como argumento del factory), `SCRIPTS` y `WORKSPACE_ROOT` de config.js
- Produces:
  - `POST /api/govern/run` → `202 { jobId }` | `400` | `409`
  - WS messages: `{ type:'output', jobId, data }` | `{ type:'done', jobId, exitCode }` | `{ type:'error', jobId, message }`

- [ ] **Step 1: Implementar govern.js**

Reemplazar `backend/routes/govern.js`:

```js
import { Router } from 'express';
import { spawn } from 'child_process';
import { createInterface } from 'readline';
import path from 'path';
import { SCRIPTS, WORKSPACE_ROOT } from '../config.js';

export default function governRouter(wss) {
  const router = Router();
  let activeJob = null;

  function broadcast(msg) {
    const data = JSON.stringify(msg);
    for (const client of wss.clients) {
      if (client.readyState === 1) client.send(data);   // 1 = WebSocket.OPEN
    }
  }

  router.post('/run', (req, res) => {
    const { script } = req.body ?? {};

    if (!script || !SCRIPTS[script]) {
      return res.status(400).json({ error: `Script desconocido: '${script}'` });
    }
    if (activeJob) {
      return res.status(409).json({ error: 'Ya hay un script corriendo' });
    }

    const jobId = `ws-${Date.now()}`;
    activeJob = jobId;
    res.status(202).json({ jobId });

    const scriptPath = path.join(WORKSPACE_ROOT, SCRIPTS[script]);

    let child;
    try {
      child = spawn('pwsh', ['-NoProfile', '-File', scriptPath], {
        cwd: WORKSPACE_ROOT,
        stdio: ['ignore', 'pipe', 'pipe'],
      });
    } catch (err) {
      broadcast({ type: 'error', jobId, message: `spawn falló: ${err.message}` });
      activeJob = null;
      return;
    }

    const rl_out = createInterface({ input: child.stdout });
    const rl_err = createInterface({ input: child.stderr });

    rl_out.on('line', (data) => broadcast({ type: 'output', jobId, data: data + '\n' }));
    rl_err.on('line', (data) => broadcast({ type: 'output', jobId, data: data + '\n' }));

    child.on('close', (code) => {
      broadcast({ type: 'done', jobId, exitCode: code ?? -1 });
      activeJob = null;
    });

    child.on('error', (err) => {
      broadcast({ type: 'error', jobId, message: err.message });
      activeJob = null;
    });
  });

  return router;
}
```

```js
const canvas = document.getElementById('particles');
const ctx    = canvas.getContext('2d');
let particles = [];

function resizeCanvas() {
  canvas.width  = window.innerWidth;
  canvas.height = window.innerHeight;
}
window.addEventListener('resize', resizeCanvas);
resizeCanvas();

window.addEventListener('mousemove', (e) => {
  for (let i = 0; i < 2; i++) {
    particles.push({
      x: e.clientX, y: e.clientY,
      vx: (Math.random() - 0.5) * 3,
      vy: -(Math.random() * 2 + 1.5),
      size: Math.random() * 2 + 2,
      life: 60, maxLife: 60,
    });
  }
});

function renderParticles() {
  ctx.clearRect(0, 0, canvas.width, canvas.height);
  particles = particles.filter(p => p.life > 0);
  for (const p of particles) {
    p.x += p.vx; p.y += p.vy; p.vy -= 0.08; p.size *= 0.97;
    const alpha = p.life / p.maxLife;
    ctx.save();
    ctx.globalAlpha = alpha;
    ctx.shadowBlur  = 8;
    ctx.shadowColor = '#00E676';
    ctx.fillStyle   = '#00E676';
    ctx.beginPath();
    ctx.arc(p.x, p.y, Math.max(p.size, 0.5), 0, Math.PI * 2);
    ctx.fill();
    ctx.restore();
    p.life--;
  }
  requestAnimationFrame(renderParticles);
}
```

```js
  for (const s of GOVERN_SCRIPTS) {
    const btn = document.createElement('button');
    btn.className = 'govern-btn';
    btn.dataset.script = s.id;
    btn.innerHTML = `<span class="govern-btn-name">${s.name}</span><span class="govern-btn-desc">${s.desc}</span>`;
    btn.addEventListener('click', () => runScript(s.id));
    grid.appendChild(btn);
  }
  document.getElementById('btn-clear')?.addEventListener('click', clearOutput);
}

function setGovBtnsDisabled(disabled) {
  document.querySelectorAll('.govern-btn').forEach(b => { b.disabled = disabled; });
}

function clearOutput() {
  document.getElementById('output-panel').innerHTML = '';
}

function appendOutput(text, cls = '') {
  const panel = document.getElementById('output-panel');
  if (!panel) return;
  const span = document.createElement('span');
  if (cls) span.className = cls;
  span.textContent = text;
  panel.appendChild(span);
  panel.scrollTop = panel.scrollHeight;
}

async function runScript(scriptId) {
  clearOutput();
  appendOutput(`$ ${scriptId}\n`);
  setGovBtnsDisabled(true);
  try {
    const res = await fetch(`${API_BASE}/api/govern/run`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ script: scriptId }),
    });
    if (!res.ok) {
      const err = await res.json();
      appendOutput((err.error || 'Error desconocido') + '\n', 'out-error');
      setGovBtnsDisabled(false);
    }
    // Si 202 OK: el output llega vía WebSocket
  } catch {
    appendOutput('Error al contactar el servidor\n', 'out-error');
    setGovBtnsDisabled(false);
  }
}

// === WebSocket client ===
let ws;

function connectWS() {
  ws = new WebSocket(`ws://${location.host}`);

  ws.onmessage = (event) => {
    const msg = JSON.parse(event.data);
    if (msg.type === 'output') {
      appendOutput(msg.data);
    } else if (msg.type === 'done') {
      if (msg.exitCode === 0) {
        appendOutput('✓ completado\n', 'out-success');
      } else {
        appendOutput(`✗ exitCode: ${msg.exitCode}\n`, 'out-error');
      }
      setGovBtnsDisabled(false);
    } else if (msg.type === 'error') {
      appendOutput(`✗ ${msg.message}\n`, 'out-error');
      setGovBtnsDisabled(false);
    }
  };

  ws.onerror = () => showError(true);

  ws.onclose = () => {
    setTimeout(connectWS, 3000);
  };
}
```
